@messages("user.name"): @{cart.userName}

@messages("user.mail"): @{cart.userMail}

@{item.drugName} @{item.num} @{item.price} @{item.price * item.num} @{item.producer} @{if(item.availableOnStock > 0) messages("yes") else messages("no")}
